

def fib(n):
	
	a = 0
	b = 1

	if n == 0:
		print(a)
	elif n == 1:
		print(a)
		print(b)
	elif type(n) != int:
		print("The no is not an integer")
	elif n < 0:
		print("invalid no, no has to be a positive no")
	else:
		print(a)
		print(b)

	for i in range(1,n):
		c = a + b
		a = b
		b = c
		
		print(c)

fib(50)
